متغير الضوء : صحيح = 9 ؛
متغير السطوع : صحيح = 0 ؛
متغير مقدار_التلاشي : صحيح = 5 ؛

دالة اعداد() : فارغ {
    وضع_الطرف(الضوء، مخرج) ؛
}

دالة تكرار() : فارغ {
    // analogWrite(led, brightness)
    اكتب_تناظري(الضوء، السطوع) ؛
    
    // استخدام معامل الجمع والإسناد
    السطوع += مقدار_التلاشي ؛
    
    // استخدام المعاملات المنطقية (أو)
    لو (السطوع <= 0 او السطوع >= 255) {
        مقدار_التلاشي = -مقدار_التلاشي ؛
    }
    
    انتظر(30) ؛
    متغير تجربة : صحيح = 100/5؛
    انتظر(تجربة)؛
}
