// تعريف منافذ المحركات والسرعة
متغير منفذ_1 : صحيح = 2 ؛
متغير منفذ_2 : صحيح = 3 ؛
متغير منفذ_3 : صحيح = 5 ؛
متغير منفذ_4 : صحيح = 4 ؛
متغير سرعة_يمين : صحيح = 9 ؛
متغير سرعة_يسار : صحيح = 10 ؛

// تعريف منافذ الإدخال (الحساسات أو الأزرار)
متغير مدخل_الخلف : صحيح = 1 ؛ // المدخل الأول (لعكس الاتجاه)
متغير مدخل_الامام : صحيح = 6 ؛ // المدخل الثاني (للتحرك بشكل مستقيم)
    متغير الحالة : صحيح = 1 ؛

دالة اعداد() : فارغ {
    // إعداد منافذ المحركات كمخارج
    وضع_الطرف(منفذ_1، مخرج) ؛
    وضع_الطرف(منفذ_2، مخرج) ؛
    وضع_الطرف(منفذ_3، مخرج) ؛
    وضع_الطرف(منفذ_4، مخرج) ؛
    وضع_الطرف(سرعة_يمين، مخرج) ؛
    وضع_الطرف(سرعة_يسار، مخرج) ؛
    وضع_الطرف(7، مخرج) ؛
        اكتب_رقمي(7، عالي) ؛
    
    // إعداد منافذ الإدخال كمداخل
    وضع_الطرف(مدخل_الخلف، مدخل) ؛
    وضع_الطرف(مدخل_الامام، مدخل) ؛
}

دالة تكرار() : فارغ {
    // قراءة حالة كلا المدخلين
    متغير حالة_الخلف : صحيح = اقرأ_رقمي(مدخل_الخلف) ؛
    متغير حالة_الامام : صحيح = اقرأ_رقمي(مدخل_الامام) ؛
    لو (حالة_الامام == منخفض){
        الحالة = 1 ؛

    }
    لو (حالة_الخلف == منخفض){
        الحالة = 0 ؛

    }
    لو (الحالة == 0) {
        // --- الشرط الأول: المدخل الأول عالي -> عكس المحركات (الرجوع للخلف) ---
        اكتب_تناظري(سرعة_يمين، 150) ؛
        اكتب_تناظري(سرعة_يسار، 150) ؛
        
        اكتب_رقمي(منفذ_1، منخفض) ؛
        اكتب_رقمي(منفذ_2، عالي) ؛
        اكتب_رقمي(منفذ_3، منخفض) ؛
        اكتب_رقمي(منفذ_4، عالي) ؛
        
    } والا {
        
            // --- الشرط الثاني: المدخل الثاني عالي -> السير للأمام (بشكل مستقيم) ---
            اكتب_تناظري(سرعة_يمين، 100) ؛
            اكتب_تناظري(سرعة_يسار، 100) ؛
            
            اكتب_رقمي(منفذ_1، عالي) ؛
            اكتب_رقمي(منفذ_2، منخفض) ؛
            اكتب_رقمي(منفذ_3، عالي) ؛
            اكتب_رقمي(منفذ_4، منخفض) ؛
            
        
        
    }
}
