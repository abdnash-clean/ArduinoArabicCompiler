متغير الليد : صحيح = 13 ؛

دالة اعداد() : فارغ {
    وضع_الطرف(الليد، مخرج) ؛
}

دالة تكرار() : فارغ {
    متغير العداد : صحيح = 0 ؛
    
    طالما (العداد < 5) {
        لو (العداد == 3) {
            // يومض بسرعة عند الرقم 3
            اكتب_رقمي(الليد، عالي) ؛
            انتظر(100) ؛
            اكتب_رقمي(الليد، منخفض) ؛
            العداد = العداد + 1 ؛
            تجاوز ؛
        } والا {
            // يومض ببطء لبقية الأرقام
            اكتب_رقمي(الليد، عالي) ؛
            انتظر(1000) ؛
            اكتب_رقمي(الليد، منخفض) ؛
            انتظر(1000) ؛
        }
        العداد = العداد + 1 ؛
    }
}
