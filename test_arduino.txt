// استدعاء مكتبة السيريال حسب الـ Registry
استيراد "سيريال" ؛

متغير العداد : صحيح = 0 ؛
متغير رسالة_ترحيب : نص = "مرحبا بك في الاردوينو" ؛
متغير النظام_يعمل : منطقي = صح ؛

// دالة مخصصة ترجع قيمة صحيحة
دالة ضاعف_الرقم(الرقم : صحيح) : صحيح {
    متغير النتيجة : صحيح = الرقم * 2 ؛
    ارجع النتيجة ؛
}

دالة اعداد() : فارغ {
    // بدء الاتصال التسلسلي بسرعة 9600
    سيريال_ابدأ(9600) ؛
    سيريال_اطبع(رسالة_ترحيب) ؛
}

دالة تكرار() : فارغ {
    طالما (النظام_يعمل == صح) {
        
        العداد++ ؛
        
        لو (العداد == 5) {
            تجاوز ؛ // continue
        }
        
        لو (العداد > 10) {
            اقطع ؛ // break
        }
        
        // استدعاء دالة داخلية وحفظ الناتج
        متغير القيمة_الجديدة : صحيح = ضاعف_الرقم(العداد) ؛
        
        // طباعة النتيجة على السيريال
        سيريال_اطبع(القيمة_الجديدة) ؛
        
        انتظر(500) ؛
    }
    
    // إيقاف التكرار (Infinite Loop) مع استخدام الزمن الحالي
    متغير الوقت : صحيح = 0 ؛
    طالما (صح) {
        الوقت = الزمن_الحالي() ؛
        انتظر(1000) ؛
    }
}
