// برنامج اختبار: وميض LED المدمج (الطرف 13 = PB5)
// يستعرض الوصول للسجلات + عوامل الإزاحة + الأرقام الثنائية

دالة اعداد():فارغ {
    // اجعل البت 5 من المنفذ ب مخرجاً: ث100000 = 32
    اتجاه_ب = ث100000؛
}

دالة تكرار():فارغ {
    // تشغيل البت 5 باستخدام الإزاحة: منفذ_ب | (1 << 5)
    منفذ_ب = منفذ_ب | (ث1 << 5)؛
    انتظر(500)؛

    // إطفاء البت 5: منفذ_ب & 223  (أي ~32 و 0xFF)
    منفذ_ب = منفذ_ب & 223؛
    انتظر(500)؛


    متغير ناتج : صحيح = ~ث11110111 & ث11111111؛   // ~247 & 0xFF = 8
}
