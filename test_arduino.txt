

دالة احسب() : صحيح {
    متغير نصي : نص = "تجربة" ؛
    
    // خطأ دلالي: نوع الإرجاع 'نص' لا يتطابق مع نوع الدالة 'صحيح'
    ارجع نصي ؛ 
}

دالة اعداد() : فارغ {
    // خطأ دلالي: نوع الإرجاع 'صحيح' لا يتطابق مع نوع الدالة 'فارغ'
    ارجع 10 ؛ 
    
    // هذا السطر صحيح لأن الدالة 'فارغ' والارجاع بدون قيمة
    ارجع ؛ 
}

دالة تكرار() : فارغ {
    متغير س : صحيح = 0 ؛
    طالما (س < 10) {
        س = س + 1 ؛
        لو (س == 5) {
            // هذا السطر صحيح لأنه داخل حلقة 'طالما'
            تجاوز ؛ 
        }
    }
}
